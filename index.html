        try {

            // ✅ Initialize the map (default view: Detroit)

            var map = L.map('map').setView([42.3314, -83.0458], 10);


            // ✅ Add OpenStreetMap tile layer

            L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {

            // ✅ Example project locations (Replace these with real data from PowerApps later)

            ];


            // ✅ Add markers for each location

                    .bindPopup("<b>" + loc.name + "</b>").openPopup();

            });


            console.log("✅ Leaflet Map Loaded Successfully");

        } catch (error) {
